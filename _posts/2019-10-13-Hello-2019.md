## 正文
<h2>python实现快排</h2>
注意的是：<br>
双指针应该先从右边扫while，再从左边扫while，这样i,j不断交换，最后在i,j重合的时候是落在那个最邻近的大于pivot的index上。然后与pivot交换就完成了挖坑填坑的快排思想。<br>

```python
#  双指针

def quick_sort_double(arr, start, end):
    if start<end:
        pivot = arr[start]
        i, j = start, end
    
        while i < j:
            while i < j and arr[j] >= pivot:
                j = j - 1
            while i < j and arr[i] <= pivot:
                i = i + 1
            if i < j:
                temp = arr[i]
                arr[i] = arr[j]
                arr[j] = temp
        arr[start] = arr[i]
        arr[i] = pivot
    
        quick_sort_double(arr, start, i - 1)
        quick_sort_double(arr, j + 1, end)


def quick_sort_single(arr, start, end):
    if start < end:
        pivot = arr[start]
        mark = start
        i, j = start, end
        while i < j:
            for index in range(i + 1, end + 1):  # 每次进来是 i+1， 不是从头开始
                if arr[index] < pivot:
                    mark = mark + 1
                    temp = arr[mark]
                    arr[mark] = arr[index]
                    arr[index] = temp
            i = index  # 赋值给i控制循环结束
            arr[start] = arr[mark]
            arr[mark] = pivot

        quick_sort_single(arr, start, mark - 1)
        quick_sort_single(arr, mark + 1, end)


arr = [2, 1, 5, 6, 4, 3, 7, 8]
quick_sort_single(arr, 0, 7)
print(arr)

```

<h2>Java实现快排</h2>
```java
package 排序;

import java.util.ArrayList;

public class QuickSort {
	int array[];
	QuickSort(int  array[]) {
		this.array = array;
	}
	//递归调用，双指针
	public void quick_sort(int arr[], int start,int end) {
		if(start<end) {
			int pivot = array[start];
			int i = start;
			int j = end;
			
			while(i<j) {
				while(i<j && array[j]>pivot) {
					j--;
				}
				while(i<j && array[i]<=pivot) {
					i++;
				}
				if(i<j) {
					int temp = array[i];
					array[i] = array[j];
					array[j] = temp; 
				}
			}
			array[start] = array[i];
			array[i] = pivot;
			quick_sort(array,start,i-1); //注意减号
			quick_sort(array,j+1,end);
			
		}
		
		
	}	
	
	public void quick_sort_single(int arr[], int start, int end) {
		if(start<end) {
			int pivot = arr[start];
			int i = start;
			int j = end;
			int mark = start;
			int index;
			if(i<j) {
				for(index=i+1;index<=j;index++) {
					if(arr[index]<pivot) {
						mark++;
						int temp = arr[mark];
						arr[mark] = arr[index];
						arr[index] = temp;
					}
				}
				arr[start] = arr[mark];
				arr[mark] = pivot;
			}
			quick_sort_single(arr,start,mark-1);
			quick_sort_single(arr,mark+1,end);


		}
		
		
		
		
	}
	public static void main(String[] args) {
		int arr[]={2,1,5,3,4,6,7,8};
		QuickSort quick = new QuickSort(arr);
		//quick.quick_sort(arr,0, 7);	
		quick.quick_sort_single(arr, 0, 7);
		for(int i=0;i<arr.length;i++) {
			System.out.println(arr[i]);
		}
	}
}




```
